% Minimal Template - Clean, Modern, Lots of Whitespace
% For Pandoc conversion of Markdown to PDF

\documentclass[12pt,letterpaper]{article}

% Packages
\usepackage[margin=1.25in]{geometry}
\usepackage{titlesec}
\usepackage{graphicx}
\usepackage{xcolor}
\usepackage{enumitem}
\usepackage{hyperref}
\usepackage{parskip}

% Font setup - Latin Modern
\usepackage{lmodern}

% Colors
\definecolor{darkgray}{RGB}{64,64,64}

% Remove section numbering
\setcounter{secnumdepth}{0}

% Section formatting - simple, minimal
\titleformat{\section}
  {\normalfont\Large\bfseries\color{darkgray}}{\thesection}{1em}{}
\titleformat{\subsection}
  {\normalfont\large\bfseries}{\thesubsection}{1em}{}
\titleformat{\subsubsection}
  {\normalfont\normalsize\bfseries}{\thesubsubsection}{1em}{}

% No headers - just centered page numbers in footer
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\fancyfoot[C]{\small\thepage}

% Hyperlink setup - minimal
\hypersetup{
    colorlinks=true,
    linkcolor=black,
    urlcolor=darkgray,
    pdfborder={0 0 0}
}

% Pandoc variables
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Title block - simple and left-aligned
\title{
    \vspace{-3em}
    \begin{flushleft}
    {\Huge\bfseries $title$}\\[1em]
    $if(author)${\large $author$}\\[0.3em]$endif$
    $if(date)${\normalsize $date$}$endif$
    \end{flushleft}
    \vspace{1em}
}
\author{}
\date{}

\begin{document}

\maketitle
\thispagestyle{fancy}

$body$

\end{document}
